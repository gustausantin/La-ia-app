            .from('availability_slots')

            .eq('slot_date', reservation.reservation_date)
            .eq('start_time', reservation.reservation_time);

        if (slotError) {
            console.warn('‚ö†Ô∏è No se pudieron liberar los slots:', slotError);
        }
